02039     // -----------------------------------------------------------------------
02040     //  Private data members
02041     //
02042     //  fAttrList
02043     //      A temporary implementation of the basic SAX attribute list
02044     //      interface. We use this one over and over on each startElement
02045     //      event to allow SAX-like access to the element attributes.
02046     //
02047     //  fDocHandler
02048     //      The installed SAX doc handler, if any. Null if none.
02049     //
02050     //  fDTDHandler
02051     //      The installed SAX DTD handler, if any. Null if none.
02052     //
02053     //  fElemDepth
02054     //      This is used to track the element nesting depth, so that we can
02055     //      know when we are inside content. This is so we can ignore char
02056     //      data outside of content.
02057     //
02058     //  fEntityResolver
02059     //      The installed SAX entity handler, if any. Null if none.
02060     //
02061     //  fErrorHandler
02062     //      The installed SAX error handler, if any. Null if none.
02063     //
02064     //  fPSVIHandler
02065     //      The installed PSVI handler, if any. Null if none.
02066     //
02067     //  fAdvDHCount
02068     //  fAdvDHList
02069     //  fAdvDHListSize
02070     //      This is an array of pointers to XMLDocumentHandlers, which is
02071     //      how we see installed advanced document handlers. There will
02072     //      usually not be very many at all, so a simple array is used
02073     //      instead of a collection, for performance. It will grow if needed,
02074     //      but that is unlikely.
02075     //
02076     //      The count is how many handlers are currently installed. The size
02077     //      is how big the array itself is (for expansion purposes.) When
02078     //      count == size, is time to expand.
02079     //
02080     //  fParseInProgress
02081     //      This flag is set once a parse starts. It is used to prevent
02082     //      multiple entrance or reentrance of the parser.
02083     //
02084     //  fScanner
02085     //      The scanner being used by this parser. It is created internally
02086     //      during construction.
02087     //
02088     //   fGrammarPool
02089     //      The grammar pool passed from external application (through derivatives).
02090     //      which could be 0, not owned.
02091     //
02092     // -----------------------------------------------------------------------
02093     bool                 fParseInProgress;
02094     unsigned int         fElemDepth;
02095     unsigned int         fAdvDHCount;
02096     unsigned int         fAdvDHListSize;
02097     VecAttrListImpl      fAttrList;
02098     DocumentHandler*     fDocHandler;
02099     DTDHandler*          fDTDHandler;
02100     EntityResolver*      fEntityResolver;
