=head1 NAME

XmlTransaction - A Perl interface to the DbXml XmlTransaction Class

=head1 SYNOPSIS

    use Sleepycat::DbXml;

    my $mgr = new XmlManager;

    $txn = mgr->createTransaction;

    $txn->abort();
    $txn->commit([$flags]);
    my $txn2 = $txn->createChild([$flags]);
    my $dbtxn = $txn->getDbTxn();
